    "Tangle config.org automatically if in `user-emacs-directory`"
    (when (string-equal (buffer-file-name)
                        (expand-file-name "config.org" user-emacs-directory))
      (let ((org-confirm-babel-evaluate nil))
        (org-babel-tangle))))
